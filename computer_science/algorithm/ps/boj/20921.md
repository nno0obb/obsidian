def main():
    # Input
    N, K = map(int, input().split())

    # Logic
    ans, lst = [], []
    for k in range(N - 1, -1, -1):
        n = k + 1
        if K >= k:
            ans.append(n)
            K -= k
        else:
            lst.append(n)
    ans += lst[::-1]

    # Output
    print(*ans)


if __name__ == "__main__":
    main()
